68 public String changeFileAcl() {
69 if(!newPermissions.matches("[rwx]+")) {
70 errMsg.replace(0, errMsg.length(), "Error: Invalid permissions:").append(newPermissions);
71 return errMsg.toString();
72 }
73 UserPrincipal newOwner = getUserPrinciple(targetFilePath, newOwnerUsername);
74 if (newOwner == null) {

78
79 // Try to change the owner of the file
80 String newOwnerChangeResults = setNewOwner(newOwner);
81 if (newOwnerChangeResults.contains("Error")) {
82 return newOwnerChangeResults;

84

88 return newPermissionsChangeResults;
89 }

91 // If we get here, both the owner and permissions were changed successfully
92 String successMessage = String.format("Success: Changed owner and permissions of file %s", targetFilePath.toAbsolutePath());
93 logFine(successMessage);
94 return successMessage;
95 }

144 // Change the permissions of the file using ACL on Windows OS
145 private String setNewPermissionsWindows(UserPrincipal newOwner) {
146 try {
147 String statusMessage;
148 // Change the permissions of the file using ACLs
149 AclFileAttributeView aclView = Files.getFileAttributeView(targetFilePath, AclFileAttributeView.class);
150 AclEntryPermission readPermission = AclEntryPermission.READ_DATA;
151 AclEntryPermission writePermission = AclEntryPermission.WRITE_DATA;
152 AclEntryPermission executePermission = AclEntryPermission.EXECUTE;
153
154 AclEntryType allow = AclEntryType.ALLOW;
155
156 EnumSet<AclEntryPermission> actualPermissions = EnumSet.noneOf(AclEntryPermission.class);
157 if (newPermissions.contains("r")) {
158 actualPermissions.add(readPermission);
159 }
160 if (newPermissions.contains("w")) {
161 actualPermissions.add(writePermission);
162 }
163 if (newPermissions.contains("x")) {
164 actualPermissions.add(executePermission);
165 }
166
167 statusMessage = String.format("Attempting to change permissions to %s", actualPermissions);
168 logFine(statusMessage);
169
170 // Create the new ACL entry
171 AclEntry aclEntry = AclEntry.newBuilder()
172 .setType(allow)
173 .setPrincipal(newOwner)
174 .setPermissions(actualPermissions)
175 .build();
176
177 // Get the existing ACL and add the new ACL entries
178 List<AclEntry> acl = new ArrayList<>();
179 acl.add(aclEntry);
180
181 // Set the new ACL
182 aclView.setAcl(acl);
183 errMsg.replace(0, errMsg.length(), "Success: Changed permissions of file").append(targetFilePath.toAbsolutePath()).append(" to ").append(actualPermissions);
184 logFine(errMsg.toString());
185 return errMsg.toString();
186 } catch (Exception e) {
187 errMsg.replace(0, errMsg.length(), "Error: Exception setting permissions on file with ACL: ")
188 .append(targetFilePath.toAbsolutePath())
189 .append(", ")
190 .append(e.getMessage());
191 logSevere(errMsg.toString());
