241 // Generate a pdf file with a given size in bytes
242 private byte[] generatePdfBytes(int sizeInBytes) throws Exception {
243 if (sizeInBytes <= 0) {
244 throw new IllegalArgumentException("Size in Bytes must be a positive value.");
245 }
246
247 ByteArrayOutputStream baos = new ByteArrayOutputStream();
248 Document document = new Document(PageSize.A4);
249 PdfWriter writer = PdfWriter.getInstance(document, baos);
250 writer.setFullCompression();
251 writer.setCompressionLevel(0);
252 document.open();
253
254 String content = UUID.randomUUID().toString() + UUID.randomUUID().toString();
255 int contentLength = content.getBytes().length;
256
257 while (baos.size() < sizeInBytes) {
258 int iterations = (sizeInBytes - baos.size()) / contentLength;
259 for (int i = 0; i < iterations; i++) {
260 document.add(new Paragraph(content));
261 }
262 writer.flush();
263 }
264
265 document.close();
266 writer.close();
267
268 return baos.toByteArray();
269 }

271 // Calculate file checksum
272 private static String calculateChecksum(byte[] fileBytes) {
273 try {
274 byte[] hash = MessageDigest.getInstance("SHA-256").digest(fileBytes);
275 return new BigInteger(1, hash).toString(16);
276 } catch (Exception e) {
277 e.printStackTrace();
278 return null;
