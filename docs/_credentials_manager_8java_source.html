89 public char[] getCredentials(String credsName) throws IllegalArgumentException, CredentialsFetchError {
90 if (credsName == null || credsName.isEmpty()) {
91 logSevere(logMessage.replace(0, logMessage.length() ,"Credentials name was empty."));
92 throw new IllegalArgumentException("Credentials name was empty.");
93 }
94 if (!this.credsNamesList.contains(credsName)) {
95 logSevere(logMessage.replace(0, logMessage.length() ,"Credentials name was not found in credentials list."));

108 logFine(logMessage.replace(0, logMessage.length() ,"Adding ").append(credsName).append(" to credentials list."));
109 Path fullCredPath = this.credsPath.resolve(credsName);
110 credsSynchronizer.addItem(addSalt(this.getCredFileChars(fullCredPath)), credsName);

113 } catch (Exception e) {

115 throw new CredentialsFetchError("Error retrieving credentials: " + credsName);

119 /**
120 * This method is used to read a single credential file into a char[] for storage in the Map.
121 * char[] grows dynamically as the file is read using 64 byte chunks.
122 * @param credFilePath (Path) - Path to the credentials file to read.
123 * @return (char[]) - The credentials as a char[].
124 * @throws CredentialsFetchError - If there is an error reading the credentials file, or if the credentials file is empty.
125 */
126 private char[] getCredFileChars(Path credFilePath) throws CredentialsFetchError {
127 logFine(logMessage.replace(0, logMessage.length() ,"Reading credentials from file: ").append(credFilePath.getFileName()));
128 FileReader fileHandle;
129 try {
130 fileHandle = new FileReader(credFilePath.toFile());
131 } catch (FileNotFoundException e) {
132 throw new CredentialsFetchError("Credentials file was not found: " + credFilePath.getFileName());
133 }
134 try (Reader br = new SecureFileReader(fileHandle, 1)) {
135 char[] tmpCharArray = new char[1];
136 int charRead = 0;
137 int position = 0;
138 while((charRead = br.read()) != -1) {
139 if(position == tmpCharArray.length) {
140 char[] temp = new char[tmpCharArray.length + 1];
141 System.arraycopy(tmpCharArray, 0, temp, 0, position);
142 tmpCharArray = temp;
143 }
144 tmpCharArray[position++] = (char) charRead;
145 }
146 logFine(logMessage.replace(0, logMessage.length() ,"Credentials read from file: ").append(credFilePath.toString()));
147 if (tmpCharArray[0] == '\u0000') {
148 throw new CredentialsFetchError("Credentials file was empty: " + credFilePath.getFileName());
149 }
150 return tmpCharArray;
151 } catch (Exception e) {
152 throw new CredentialsFetchError("Error reading credentials file: " + e.getMessage());
153 } finally {
154 try {
155 fileHandle.close();
156 } catch (Exception e) {
157 logSevere(logMessage.replace(0, logMessage.length() ,"Error closing credentials file: ").append(e.getMessage()));
158 }
159 }
160 }

210 private void generateRandomString() {
211 int leftLimit = 48; // numeral '0'
212 int rightLimit = 122; // letter 'z'
213 int targetStringLength = 2048;
214 this.randomString = random.ints(leftLimit, rightLimit + 1)
215 .filter(i -> (i <= 57 || i >= 65) && (i <= 90 || i >= 97))
216 .limit(targetStringLength)
217 .collect(StringBuilder::new, StringBuilder::appendCodePoint, StringBuilder::append)
218 .toString();
219 logFine(logMessage.replace(0, logMessage.length() ,"Random string generated."));

222 /**
223 * Add unique salt to a given string.
224 * @param charString (char[]) - The string to add salt to.
225 * @return (char[]) - The salted string.
226 */
227 private char[] addSalt(char[] charString) {
228 if (this.randomStringIndex + charString.length > this.randomString.length()) {
229 generateRandomString();
230 this.randomStringIndex = 0;
231 }
232 char[] saltChars = randomString.substring(randomStringIndex, randomStringIndex + charString.length).toCharArray();
233 char[] saltedString = new char[charString.length * 2];
234 int index = 0;
235 for (int i = 0; i < charString.length; i++) {
236 saltedString[index++] = charString[i];
237 saltedString[index++] = saltChars[i];
238 }
239 this.randomStringIndex += saltChars.length;
240 return saltedString;
241 }
