128 private String getPosixFilePermissions (Path path) throws Exception {
129 // Get the Posix file permissions for the file
130 Set<PosixFilePermission> permissions = Files.getPosixFilePermissions(path);
131
132 // Convert the user permissions to rwx format
133 StringBuilder rwxPermissions = new StringBuilder();
134 for (PosixFilePermission permission : EnumSet.copyOf(permissions)) {
135 switch (permission) {
136 case OWNER_READ:
137 rwxPermissions.append("r");
138 break;
139 case OWNER_WRITE:
140 rwxPermissions.append("w");
141 break;
142 case OWNER_EXECUTE:
143 rwxPermissions.append("x");
144 break;
145 default:
146 break;
147 }
148 }
149
150 return rwxPermissions.toString();
151 }

153 private String getAclPermissions (Path path) throws Exception {
154 // Get the ACL entries for the tmp test file
155 AclFileAttributeView view = Files.getFileAttributeView(
156 Paths.get(path.toUri()), AclFileAttributeView.class);
157
158 // Get the ACL entries for the file
159 String permissions = "";
160 for (AclEntry entry : view.getAcl()) {
161 // Check each permission and add the corresponding symbol to the output string
162 if (entry.permissions().contains(AclEntryPermission.READ_DATA)) {
163 permissions += "r";
164 }
165 if (entry.permissions().contains(AclEntryPermission.WRITE_DATA)) {
166 permissions += "w";
167 }
168 if (entry.permissions().contains(AclEntryPermission.EXECUTE)) {
169 permissions += "x";
170 }
171 }
172 return permissions.replace(String.valueOf("-"), "" );
173 }
