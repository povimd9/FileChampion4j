156 // Generate a pdf file with a given size in bytes
157 private byte[] generatePdfBytes(int sizeInBytes) throws Exception {
158 if (sizeInBytes <= 0) {
159 throw new IllegalArgumentException("Size in Bytes must be a positive value.");
160 }
161
162 ByteArrayOutputStream baos = new ByteArrayOutputStream();
163 Document document = new Document(PageSize.A4);
164 PdfWriter writer = PdfWriter.getInstance(document, baos);
165 writer.setFullCompression();
166 writer.setCompressionLevel(0);
167 document.open();
168
169 String content = UUID.randomUUID().toString() + UUID.randomUUID().toString();
170 int contentLength = content.getBytes().length;
171
172 while (baos.size() < sizeInBytes) {
173 int iterations = (sizeInBytes - baos.size()) / contentLength;
174 for (int i = 0; i < iterations; i++) {
175 document.add(new Paragraph(content));
176 }
177 writer.flush();
178 }
179
180 document.close();
181 writer.close();
182
183 return baos.toByteArray();
184 }
