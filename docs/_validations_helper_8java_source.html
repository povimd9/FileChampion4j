275 /**
276 * Check if the file contains the expected magic bytes
277 * @param originalFileBytes (byte[]) the file bytes of the file being validated
278 * @param magicBytesPattern (String) the expected magic bytes of the file being validated
279 * @return Boolean (Boolean) true if the file contains the expected magic bytes, false otherwise
280 */
281 private boolean containsMagicBytesProcessor(byte[] originalFileBytes, String magicBytesPattern) throws NumberFormatException{
282 if (originalFileBytes.length == 0 || magicBytesPattern == null || magicBytesPattern.isEmpty()) {
283 return false;
284 }
285 magicBytesPattern = magicBytesPattern.replaceAll("\\p{Zs}", "");
286 if (magicBytesPattern.length() % 2 != 0) {
287 magicBytesPattern = "0" + magicBytesPattern;
288 }
289 byte[] magicBytes = new byte[magicBytesPattern.length() / 2];
290 for (int i = 0; i < magicBytesPattern.length(); i += 2) {
291 magicBytes[i / 2] = (byte) Integer.parseInt(magicBytesPattern.substring(i, i + 2), 16);
292 }
293 for (int i = 0; i < originalFileBytes.length - magicBytes.length; i++) {
294 boolean found = true;
295 for (int j = 0; j < magicBytes.length; j++) {
296 if (originalFileBytes[i + j] != magicBytes[j]) {
297 found = false;
298 break;
299 }
300 }
301 if (found) {
302 return true;
303 }
304 }
305 return false;
306 }
