24 /**
25 * Returns a list of PluginConfig objects.
26 * @return (List<PluginConfig>) a list of PluginConfig objects
27 */
28 public Map<String, PluginConfig> getPluginConfigs() {
29 HashMap<String, PluginConfig> pluginConfigs = new HashMap<>();
30 Map<String, StepConfig> stepConfigs = new HashMap<>();
31
32
33 for (String pluginName : plugins.keySet()) {
34 JSONObject plugin = plugins.getJSONObject(pluginName);
35 PluginConfig pluginConfig = new PluginConfig();
36 pluginConfig.setName(pluginName);
37
38 for (String stepName : plugin.keySet()) {
39 JSONObject step = plugin.getJSONObject(stepName);
40 StepConfig stepConfig = new StepConfig();
41 String pluginStepName = pluginName + "." + stepName.substring(0, stepName.lastIndexOf('.'));
42 stepConfig.setName(pluginStepName);
43
44 stepConfig.setType(step.getString("type"));
45 stepConfig.setRunBefore(step.optBoolean("run_before"));
46 stepConfig.setRunAfter(step.optBoolean("run_after"));
47 stepConfig.setCredsPath(step.optString("creds_path"));
48 stepConfig.setTimeout(step.getInt("timeout"));
49 stepConfig.setOnFail(step.getString("on_timeout_or_fail"));
50 stepConfig.setEndpoint(step.getString("endpoint"));
51 stepConfig.setResponse(step.getString("response"));
52
53 switch (stepConfig.getType()) {
54 case "cli":
55 CliPluginHelper cliPluginStep = new CliPluginHelper(stepConfig);
56 stepConfig.setCliPluginHelper(cliPluginStep);
57 stepConfigs.put(pluginStepName, stepConfig);
58
59 continue;
60 case "http":
61 stepConfig.setMethod(step.optString("method"));
62 stepConfig.setHeaders(step.optJSONObject("headers"));
63 stepConfig.setBody(step.optJSONObject("body"));
64 stepConfig.setHttpPassCode(step.optInt("http_pass_code"));
65 stepConfig.setHttpFailCode(step.optInt("http_fail_code"));
66 stepConfigs.put(pluginStepName, stepConfig);
67 continue;
68 default:
69 continue;
70 }
71 }
72 pluginConfig.setStepConfigs(stepConfigs);
73 pluginConfigs.put(pluginName, pluginConfig);
74 }
75 return pluginConfigs;
76 }
