90 /**
91 * This method is used to set the values of the Extension object from the Json configuration data.
92 * @param configJsonObject (JSONObject) the Json configuration data
93 * @throws IllegalArgumentException if the file category or file extension is not found in the Json configuration data
94 */
95 private void setValuesFromJson(JSONObject configJsonObject) {
96 if (!configJsonObject.has(fileCategory)) {
97 throw new IllegalArgumentException("Category not found in JSON: " + fileCategory);
98 }
99 JSONObject categoryJson = configJsonObject.optJSONObject(fileCategory);
100
101 if (!categoryJson.has(fileExtension)) {
102 throw new IllegalArgumentException("Extension not found in JSON: " + fileExtension);
103 }
104 JSONObject extensionJson = categoryJson.optJSONObject(fileExtension);
105
106 this.extensionPlugins = extensionJson.optJSONArray("extension_plugins");
107 this.mimeType = extensionJson.optString("mime_type");
108 this.magicBytes = extensionJson.optString("magic_bytes");
109 this.headerSignatures = extensionJson.optString("header_signatures");
110 this.footerSignatures = extensionJson.optString("footer_signatures");
111 this.antivirusScanJson = extensionJson.optJSONArray("antivirus_scan");
112 this.changeOwnership = extensionJson.optBoolean("change_ownership");
113 this.changeOwnershipUser = extensionJson.optString("change_ownership_user");
114 this.changeOwnershipMode = extensionJson.optString("change_ownership_mode");
115 this.nameEncoding = extensionJson.optBoolean("name_encoding");
116 this.maxSize = extensionJson.optInt("max_size");
117 }
