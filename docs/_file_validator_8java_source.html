271 private ValidationResponse validateFileMain() {
272 commonLogString = String.format(" for file extension: %s", fileExtension);
273 responseAggregationFail = "";
274 responseMsgCountFail = 0;
275 sbresponseAggregationFail = new StringBuilder(responseAggregationFail);
276 responseAggregationSuccess = "";
277 responseMsgCountSuccess = 0;
278 sbresponseAggregationSuccess = new StringBuilder(responseAggregationSuccess);
279
280 // Check that the input parameters are not null or empty
281 checkMethodInputs(fileCategory, originalFile, fileName);
282
283 // Initialize variables
284 fileExtension = getFileExtension(fileName);
285 String originalFilenameClean = fileName.replaceAll("[^\\p{IsAlphabetic}\\p{IsDigit}.]", "_");
286
287 // Log the file type category being validated
288 sharedMessage.replace(0, sharedMessage.length(), "Validating ").append(originalFilenameClean).append(", as file type: ").append(fileCategory);
289 logInfo(sharedMessage.toString());
290
291 // Check for before plugins
292 if (extensions.getValidationValue(fileCategory, fileExtension, "extension_plugins") != null) {
293 String executionResults = executeBeforePlugins(fileCategory, fileExtension);
294 if (executionResults.contains(". Failed for step:")) {
295 sharedMessage.replace(0, sharedMessage.length(), "executeBeforePlugins failed for file: ").append(originalFilenameClean).append(", Results: ").append(executionResults);
296 logWarn(sharedMessage.toString());
297 return new ValidationResponse(false, errorResponse, sharedMessage.toString() , originalFilenameClean, null, null);
298 } else if (executionResults.contains(". Error for step:")) {
299 sharedMessage.replace(0, sharedMessage.length(), "Error executing Plugins defined to run before validations for file: ").append(originalFilenameClean).append(", Results: ").append(executionResults);
300 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
301 .append("executeBeforePlugins passed with error: ")
302 .append(executionResults);
303 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
304 logFine(responseAggregationSuccess);
305 logWarn("Error in executeBeforePlugins: " + executionResults);
306 } else {
307 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
308 .append("executeBeforePlugins executed successfully: ")
309 .append(executionResults);
310 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
311 logFine(responseAggregationSuccess);
312 logInfo(executionResults);
313 }
314 } else {
315 sharedMessage.replace(0, sharedMessage.length(), "No before plugins defined for file: ").append(originalFilenameClean);
316 logInfo(sharedMessage.toString());
317 }
318 return (doValidations(originalFilenameClean));

321 /**
322 * Following initial validations and before plugins, this method is used to execute the validations for the file.
323 * @param originalFilenameClean (String) a string containing the cleaned file name
324 * @return ValidationResponse (ValidationResponse) a ValidationResponse object containing the results of the validation
325 * @throws IllegalArgumentException - If any of the required inputs are null or empty.
326 */
327 private ValidationResponse doValidations(String originalFilenameClean) {
328 // Check that the file size is not greater than the maximum allowed size, dont continue if it is
329 int maxSize;
330 try {
331 maxSize = Integer.parseInt(extensions.getValidationValue(fileCategory, fileExtension, "max_size") != null ? extensions.getValidationValue(fileCategory, fileExtension, "max_size").toString() : "-1");
332 } catch (NumberFormatException e) {
333 maxSize = -1;
334 }
335 if (maxSize > -1 && Boolean.FALSE.equals(checkFileSize(originalFile.length, maxSize))) {
336 sbresponseAggregationFail.append(System.lineSeparator() + ++responseMsgCountFail + ". ")
337 .append("File size (")
338 .append(originalFile.length / 1000)
339 .append("KB) exceeds maximum allowed size (")
340 .append(maxSize)
341 .append("KB)")
342 .append(commonLogString);
343 responseAggregationFail = sbresponseAggregationFail.toString();
344 logWarn(responseAggregationFail);
345 return new ValidationResponse(false, errorResponse, responseAggregationFail, originalFilenameClean, originalFile, calculateChecksum(originalFile));
346 } else {
347 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
348 .append("File size check passed, file size: ")
349 .append(originalFile.length / 1000)
350 .append("KB");
351 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
352 logFine(responseAggregationSuccess);
353 }
354
355 // Check that the mime type is allowed
356 String mimeType = (String) extensions.getValidationValue(fileCategory, fileExtension, "mime_type");
357 if (!isBlank(mimeType) && !checkMimeType(originalFile, fileExtension, mimeType)) {
358 sbresponseAggregationFail.append(System.lineSeparator() + ++responseMsgCountFail + ". ")
359 .append("Invalid mime_type")
360 .append(commonLogString);
361 responseAggregationFail = sbresponseAggregationFail.toString();
362 logWarn(responseAggregationFail);
363 } else {
364 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
365 .append("Mime type check passed, mime type: ")
366 .append(mimeType);
367 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
368 logFine(responseAggregationSuccess);
369 }
370
371 // Check that the file contains the magic bytes
372 String magicBytes = (String) extensions.getValidationValue(fileCategory, fileExtension, "magic_bytes");
373 if (!isBlank(magicBytes) && !containsMagicBytes(originalFile, magicBytes)) {
374 sbresponseAggregationFail.append(System.lineSeparator() + ++responseMsgCountFail + ". ")
375 .append("Invalid magic_bytes")
376 .append(commonLogString);
377 responseAggregationFail = sbresponseAggregationFail.toString();
378 logWarn(responseAggregationFail);
379 } else {
380 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
381 .append("Magic bytes check passed, magic bytes: ")
382 .append(magicBytes);
383 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
384 logFine(responseAggregationSuccess);
385 }
386
387 // Check header signatures
388 String headerSignatures = (String) extensions.getValidationValue(fileCategory, fileExtension, "header_signatures");
389 if (!isBlank(headerSignatures) && !containsHeaderSignatures(originalFile, headerSignatures)) {
390 sbresponseAggregationFail.append(System.lineSeparator() + ++responseMsgCountFail + ". ")
391 .append("Invalid header_signatures")
392 .append(commonLogString);
393 responseAggregationFail = sbresponseAggregationFail.toString();
394 logWarn(responseAggregationFail);
395 } else {
396 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
397 .append("Header signatures check passed, header signatures: ")
398 .append(headerSignatures);
399 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
400 logFine(responseAggregationSuccess);
401 }
402
403 // Check footer signatures
404 String footerSignatures = (String) extensions.getValidationValue(fileCategory, fileExtension, "footer_signatures");
405 if (!isBlank(footerSignatures) && !containsFooterSignatures(originalFile, footerSignatures)) {
406 sbresponseAggregationFail.append(System.lineSeparator() + ++responseMsgCountFail + ". ")
407 .append("Invalid footer_signatures")
408 .append(commonLogString);
409 responseAggregationFail = sbresponseAggregationFail.toString();
410 logWarn(responseAggregationFail);
411 } else {
412 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
413 .append("Footer signatures check passed, footer signatures: ")
414 .append(footerSignatures);
415 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
416 logFine(responseAggregationSuccess);
417 }
418
419
420 // Check for after plugins
421 if (extensions.getValidationValue(fileCategory, fileExtension, "extension_plugins") != null) {
422 String executionResults = executeAfterPlugins(fileCategory, fileExtension);
423 if (executionResults.contains(". Failed for step:")) {
424 sbresponseAggregationFail.append(System.lineSeparator() + ++responseMsgCountFail + ". ")
425 .append("Error in executeAfterPlugins: ")
426 .append(executionResults)
427 .append(commonLogString);
428 responseAggregationFail = sbresponseAggregationFail.toString();
429 logWarn(responseAggregationFail);
430 } else if (executionResults.contains(". Error for step:")) {
431 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
432 .append("executeAfterPlugins passed with error: ")
433 .append(executionResults);
434 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
435 logFine(responseAggregationSuccess);
436 logWarn("Error in executeAfterPlugins: " + executionResults);
437 } else {
438 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
439 .append("executeAfterPlugins executed successfully: ")
440 .append(executionResults);
441 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
442 logFine(responseAggregationSuccess);
443 }
444 } else {
445 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
446 .append("No after plugins to execute");
447 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
448 logFine(responseAggregationSuccess);
449 }
450
451 // Check if file passed all defined validations, return false and reason if not.
452 if (responseMsgCountFail > 0) {
453 logFine(responseAggregationFail);
454 return new ValidationResponse(false, errorResponse, responseAggregationFail, originalFilenameClean, originalFile, calculateChecksum(originalFile));
455 }
456
457 // Check if the file name should be encoded
458 String encodedFileName = "";
459 boolean isNameEncoding = extensions.getValidationValue(fileCategory, fileExtension, "name_encoding") != null ? (boolean) extensions.getValidationValue(fileCategory, fileExtension, "name_encoding") : false;
460 if (isNameEncoding) {
461 sharedMessage.replace(0, sharedMessage.length(), Base64.getEncoder().encodeToString(originalFilenameClean.getBytes(StandardCharsets.UTF_8))).append(".").append(fileExtension);
462 encodedFileName = sharedMessage.toString();
463 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
464 .append("File name: ")
465 .append(originalFilenameClean)
466 .append(" has been successfully encoded to: ")
467 .append(encodedFileName);
468 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
469 logFine(responseAggregationSuccess);
470 }
471 String targetFileName = encodedFileName.isEmpty() ? originalFilenameClean : encodedFileName;
472
473 // Check if the file should be saved to output directory
474 String savedFilePath;
475 if (outDir != null && !isBlank(outDir.toString())) {
476 savedFilePath = saveFileToOutputDir(fileCategory, fileExtension, outDir, targetFileName, originalFile);
477 if (savedFilePath.contains("Error:")) {
478 // Return valid file response if file failed to save to output directory
479 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
480 .append("File is valid but failed to save to output directory: ")
481 .append(savedFilePath);
482 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
483 logInfo(responseAggregationSuccess);
484 return new ValidationResponse(true, "File is valid but failed to save to output directory", responseAggregationSuccess, originalFilenameClean, originalFile, calculateChecksum(originalFile));
485 }
486 // Return valid file response if file was saved to output directory
487 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
488 .append("File is valid and was saved to output directory: ")
489 .append(savedFilePath);
490 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
491 logInfo(responseAggregationSuccess);
492 return new ValidationResponse(true, "File is valid and was saved to output directory", responseAggregationSuccess, originalFilenameClean, originalFile, calculateChecksum(originalFile), savedFilePath);
493 }
494
495 // Return valid response if file passed all validations but is not meant to be saved to disk
496 sbresponseAggregationSuccess.append(System.lineSeparator() + ++responseMsgCountSuccess + ". ")
497 .append("File is valid: ")
498 .append(originalFilenameClean);
499 responseAggregationSuccess = sbresponseAggregationSuccess.toString();
500 logInfo(responseAggregationSuccess);
501 return new ValidationResponse(true, "File is valid", responseAggregationSuccess, originalFilenameClean, originalFile, calculateChecksum(originalFile));
502 }

562 /**
563 * Method to check that all extensions defined in config exist in plugins configuration from Validation.json
564 */
565 private void checkPluginsConfig() {
566 // check that all extensions defined plugins exist in maps
567 JSONObject validationsJsonObject = configJsonObject.getJSONObject("Validations");
568 for (int i = 0; i < validationsJsonObject.length(); i++) {
569 String categroyKey = validationsJsonObject.names().getString(i);
570 for (int k=0; k < validationsJsonObject.getJSONObject(categroyKey).length(); k++) {
571 String extensionKey = validationsJsonObject.getJSONObject(categroyKey).names().getString(k);
572 if (validationsJsonObject.getJSONObject(categroyKey).getJSONObject(extensionKey).has("extension_plugins")) {
573 checkPluginsExist(validationsJsonObject, categroyKey, extensionKey);
574 }
575 }
576 }

766 sharedMessage.replace(0, sharedMessage.length(), sharedStepMessage).append(extensionPluginName)

775 // TODO: Implement http plugin type

878 /**
879 * Check if the file contains the expected magic bytes
880 * @param originalFileBytes (byte[]) the file bytes of the file being validated
881 * @param magicBytesPattern (String) the expected magic bytes of the file being validated
882 * @return Boolean (Boolean) true if the file contains the expected magic bytes, false otherwise
883 */
884 private boolean containsMagicBytes(byte[] originalFileBytes, String magicBytesPattern) {
885 if (originalFileBytes.length == 0 || magicBytesPattern == null || magicBytesPattern.isEmpty()) {
886 return false;
887 }
888 magicBytesPattern = magicBytesPattern.replaceAll("\\p{Zs}", "");
889 if (magicBytesPattern.length() % 2 != 0) {
890 magicBytesPattern = "0" + magicBytesPattern;
891 }
892 byte[] magicBytes = new byte[magicBytesPattern.length() / 2];
893 for (int i = 0; i < magicBytesPattern.length(); i += 2) {
894 magicBytes[i / 2] = (byte) Integer.parseInt(magicBytesPattern.substring(i, i + 2), 16);
895 }
896 for (int i = 0; i < originalFileBytes.length - magicBytes.length; i++) {
897 boolean found = true;
898 for (int j = 0; j < magicBytes.length; j++) {
899 if (originalFileBytes[i + j] != magicBytes[j]) {
900 found = false;
901 break;
902 }
903 }
904 if (found) {
905 return true;
906 }
907 }
908 return false;
909 }
